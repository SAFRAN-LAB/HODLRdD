64 template<int N, class Variant, class Archive>
65 typename std::enable_if<N < std::variant_size_v<Variant>, void>::type
66 load_variant(Archive & ar, int target, Variant & variant)
67 {
68 if(N == target)
69 {
70 variant.template emplace<N>();
71 ar( CEREAL_NVP_("data", std::get<N>(variant)) );
72 }
73 else
74 load_variant<N+1>(ar, target, variant);
75 }
